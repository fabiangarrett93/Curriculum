Different values of K

# Different values of K originate different clusters

{
  fit.kmeans.1 <- kmeans(wholesale.std, centers = 1, nstart = 20)
  fit.kmeans.2 <- kmeans(wholesale.std, centers = 2, nstart = 20)
  fit.kmeans.3 <- kmeans(wholesale.std, centers = 3, nstart = 20)
  fit.kmeans.4 <- kmeans(wholesale.std, centers = 4, nstart = 20)

  # plots to compare
  p3 <- fviz_cluster(fit.kmeans.1, geom = "point", data = wholesale.std) + ggtitle("k = 1")
  p4 <- fviz_cluster(fit.kmeans.2, geom = "point", data = wholesale.std) + ggtitle("k = 2")
  p5 <- fviz_cluster(fit.kmeans.3, geom = "point", data = wholesale.std) + ggtitle("k = 3")
  p6 <- fviz_cluster(fit.kmeans.4, geom = "point", data = wholesale.std) + ggtitle("k = 4")
  
}
gridExtra::grid.arrange(p3, p4, p5, p6, nrow = 2)
